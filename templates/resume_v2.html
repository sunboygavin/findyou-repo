// 华为适配器：把通用 show 命令自动转换为厂家 display 语法
toDisplayCommand(cmd: string): string {
  const mappings: Record<string, string> = {
    'show version':            'display version',
    'show ip interface brief': 'display ip interface brief',
    'show bgp summary':        'display bgp peer',
    'show ip route':           'display ip routing-table',
    'show ospf neighbor':      'display ospf peer',
  }
  for (const [pattern, replacement] of Object.entries(mappings))
    if (cmd.toLowerCase().startsWith(pattern)) return replacement
  return cmd
}

// AI Agent 决策调用 → 经 SSH 跳板机执行 BGP 查询并结构化回传
async getBGPStatus(conn: Connection): Promise<BGPInfo[]> {
  const [output] = await this.exec(conn, ['display bgp peer'])
  return this.parseBGP(output)   // 解析后回传给 LLM 生成自然语言
}
